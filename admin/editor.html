<form data-parsley-validate>
  <div class="form-group">
    <textarea class="form-control form-summernote-custom" name="text" rows="3" required></textarea>
  </div>
  <button type="submit" class="btn btn-default">Submit</button>

(function ($) {
  'use strict';

  $('.form-summernote-custom').summernote({
    height: 320,
    minHeight: 300,
    iconPrefix: 'i i-',
    // lang: 'pt-BR',
    defaultFontName: 'Open Sans',
    toolbar: [

      ['font', ['strikethrough']],
      ['fontsize', ['fontsize']],
      ['color', ['color']],
      ['para', ['ul', 'ol', 'paragraph']],
      ['height', ['height']],
    ]
  });
}(jQuery));
